using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("¬ведите ваш рост (см)\n>");
            double Lenght = Convert.ToDouble(Console.ReadLine());
            Console.Write("¬ведите ваш вес (кг)\n>");

            double imt = Mas / (Lenght * Lenght / 100 / 100);

            Console.ReadKey();
        }
